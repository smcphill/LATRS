  def field_group_form_column(record, input_name)
    g = record.group
    if !g.nil?
      groups = Group.all(:conditions => ["id != ? and template_id = ?", 
                                         g.id, g.template_id])
      select("record", 
             "group", 
             groups.collect {|g| [g.name, g.id] })
    else
      active_scaffold_config.list.empty_field_text
    end
  end
