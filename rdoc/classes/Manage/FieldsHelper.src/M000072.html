  def field_limits_column(record)
    if record.limits.any?
      record.limits.all(:order => "position").collect {|lim|
        if lim.is_default?
          "<u><b>#{lim.name}</b></u>"
        else
          "#{lim.name}"
        end
      }.join('<br/>')
    else
      active_scaffold_config.list.empty_field_text
    end
  end
