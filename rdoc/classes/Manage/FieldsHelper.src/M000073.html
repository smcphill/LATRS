  def field_children_column(record)
    if record.children.any?
      "<ul class='children'>" + record.children.all(:order => "position").collect{|kid| 
        if kid.is_required?
          "<li style='color:red;'>#{kid.name}</li>"
        else
          "<li>#{kid.name}</li>"
        end
        }.join('') + "</ul>"
    else
      active_scaffold_config.list.empty_field_text
    end
  end
