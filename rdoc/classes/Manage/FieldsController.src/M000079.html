  def after_create_save(record)
    cond = record.group_id.nil? ? "parent_id = #{record.parent_id}" : "group_id = #{record.group_id}"
    curr_pos = Field.maximum("position", :conditions => cond)
    curr_pos ||= 0
    record.position =  curr_pos + 1
    record.save
    if record.limits.exists?
      record.limits.each do |l|
        curr_pos = nil
        curr_pos = Limit.maximum("position", :conditions => "field_id = #{record.id}")
        curr_pos ||= 0
        l.position = curr_pos + 1
        l.save
      end
    end
  end
