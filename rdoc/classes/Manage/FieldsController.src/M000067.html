  def copy
    copyfield = Field.new
    newid = copyfield.deep_copy(params[:id])
    @record = Field.find(newid)
    @src = params[:id]
    @eid = params[:eid]
    params[:adapter] = nil
    render :action => 'copy', :content_type => 'text/javascript'
  end
