  def action_links_order
    links = active_scaffold_config.action_links
    del_link = links[:delete]
    copy_link = links[:copy]
    move_link = links[:move]
    if copy_link && move_link
      links.delete :move
      links.delete :copy
      links.delete :delete
      links << copy_link
      links << move_link
      links << del_link
    end
  end
