  def auto_complete_for_patient_rn
    if params[:pid].nil?
      @p = nil
    else
      pid = params[:pid].to_s.downcase + "%"
      @p = Patient.all(:conditions => ["lower(rn) like ?", pid])
    end
      render :action => 'autocomplete'
  end
