  def rn
    if (params[:id].nil?)
      render :text => "<p>Enter the patient's RN to retrieve details</p>"
    else
      @record = Patient.find_by_rn(params[:id])
      if @record.nil?
        render :action => 'nopatient'
      else
        render :action => 'show'
      end
    end
  end
