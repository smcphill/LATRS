  def field_children(kids, type, builder, form, subtest)
    if kids.count > 0
      content_tag(:dl, :class => type) do
        s = ""
        kids.each do |k|
          s << render(:partial => "field", :locals => {:field => k, :builder => builder, :form => form, :subtest => subtest})
        end
        s
      end
    end
  end
