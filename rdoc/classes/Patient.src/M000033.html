  def age
    s = ""
    if not birthdate.nil?
      now = Date.today()
      birth = birthdate
      years = now.year - birth.year
      months = now.month - birth.month
      days = now.day - birth.day
      if days < 0
        days += 30
        months -= 1
      end
      if months < 0
        months += 12
        years -= 1
      end
      s = "#{years} yrs, #{months} mths, #{days} days"
    end
    "#{s}"
  end
