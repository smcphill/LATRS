  def before_destroy
    default = Group.find(:first, 
                         :conditions => ["template_id = ? and name = '_default'", 
                                         self.template_id])
    if (default.is_default?)
      self.fields.each do |f|
        f.group_id = default.id
        f.save
      end
      #trigger a save to (hopefully) refresh the data for AS
      default.save
    end
  end
