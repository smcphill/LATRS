      def parse_opt(op, val)
        val = val.to_s
        op = op.to_s
        case op
        when "%?%"
          op = "like"
          val = "%#{val}%"
        when "?%"
          op = "like"
          val = "#{val}%"
        when "%?"
          op = "like"
          val = "%#{val}"
        when "BETWEEN"
          return false
        end
        return [op, val]
      end
