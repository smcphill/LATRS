      def make_name(name, val, op)
        opt = parse_opt(op, val)
        if opt.kind_of?(Array)
          op = opt[0]
          val = opt[1]
          return Latrs::Report::AtomExpr.new(name, val, op)
        else
          return nil
        end
      end
