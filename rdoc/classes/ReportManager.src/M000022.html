  def malaria_strain
    malaria = Hash.new
    malaria[:name] = "Malaria: mixed strain"
    numvalue = " AND #{DB_TNUMVALS_SEARCH} >= 4"
    malaria[:desc] = "Report on the number of Malaria tests with more than one 'P' strain (including PF), where the PF RBC Infection is >= 4%\n"
    malaria[:sql] = "SELECT testables.* \nFROM testables INNER JOIN testableitems ON testableitems.testable_id = testables.id INNER JOIN testableitems as ti2 ON ti2.testable_id = testables.id AND ti2.id != testableitems.id\nWHERE testableitems.name = \"PF::RBC Infection\"\nAND (ti2.name = 'PO' or ti2.name = 'PM' or ti2.name = 'PV')\n"
    malaria[:sql] += numvalue
    return malaria
  end
